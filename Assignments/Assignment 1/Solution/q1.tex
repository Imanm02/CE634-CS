\section*{سوال ۱}

برای هر یک از سیستم‌های زیر حداقل دو نمونه برای موجودیت‌ها 
\lr{Entities}
ویژگی‌ها 
\lr{Attributes}
فعالیت‌ها 
\lr{Activities}
رویدادها 
\lr{Events}
و متغیرهای حالت 
\lr{State Variables}
نام ببرید. در بخش رویدادها، یک رویداد داخلی 
\lr{Endogenous Event}
و یک رویداد خارجی 
\lr{Exogenous Event}
باید ذکر شود.

\section*{جواب سوال ۱}

\section*{پیام‌رسان}
\begin{itemize}
	\item \textbf{موجودیت‌ها (Entities):} کاربران، پیام‌ها
	\item \textbf{ویژگی‌ها (Attributes):} نام کاربری، زمان ارسال پیام
	\item \textbf{فعالیت‌ها (Activities):} ارسال پیام، خواندن پیام
	\item \textbf{رویدادها (Events):} 
	\begin{itemize}
		\item \textbf{رویداد داخلی \lr{(Endogenous Event)}:} دریافت تاییدیه خواندن پیام
		\item \textbf{رویداد خارجی \lr{(Exogenous Event)}:} دریافت پیام جدید از یک کاربر دیگر
	\end{itemize}
	\item \textbf{متغیرهای حالت \lr{State Variables}:} تعداد پیام‌های خوانده نشده، وضعیت آنلاین یا آفلاین بودن کاربر
\end{itemize}

\section*{بانک}
\begin{itemize}
	\item \textbf{موجودیت‌ها (Entities):} حساب، مشتری
	\item \textbf{ویژگی‌ها (Attributes):} 
	\begin{itemize}
		\item حساب: شماره حساب، موجودی
		\item مشتری: نام، شناسه مشتری
	\end{itemize}
	\item \textbf{فعالیت‌ها (Activities):} واریز و برداشت پول، انتقال وجه، درخواست وام
	\item \textbf{رویدادها (Events):}
	\begin{itemize}
		\item \textbf{رویداد داخلی (Endogenous Event):} تغییر نرخ بهره بانکی
		\item \textbf{رویداد خارجی (Exogenous Event):} ورود یک چک جدید برای وصول
	\end{itemize}
	\item \textbf{متغیرهای حالت (State Variables):} تعداد کل حساب‌ها، مجموع موجودی‌های بانک
\end{itemize}

\subsection*{سایت رزرو هتل}
\begin{itemize}
	\item \textbf{موجودیت‌ها (Entities) :} مشتریان، اتاق‌های هتل
	\item \textbf{ویژگی‌ها (Attributes) :} تاریخ ورود و خروج، قیمت اتاق
	\item \textbf{فعالیت‌ها (Activities) :} جستجو برای اتاق، رزرو اتاق
	\item \textbf{رویدادها (Events):}
	\begin{itemize}
		\item \textbf{رویداد داخلی \lr{(Endogenous Event)} :} انقضای زمان رزرو اتاق
		\item \textbf{رویداد خارجی \lr{(Exogenous Event)} :} تغییر قیمت اتاق به دلیل تغییرات فصلی
	\end{itemize}
	\item \textbf{متغیرهای حالت \lr{(State Variables)} :} تعداد اتاق‌های موجود، تعداد رزروهای انجام‌شده
\end{itemize}

\subsection*{کارخانه‌ی ساخت مبل}
\begin{itemize}
	\item \textbf{موجودیت‌ها (Entities) :} خط تولید، مبل‌ها
	\item \textbf{ویژگی‌ها (Attributes) :} مدل مبل، زمان تولید
	\item \textbf{فعالیت‌ها (Activities) :} برش چوب، روکش‌کاری
	\item \textbf{رویدادها (Events) :}
	\begin{itemize}
		\item \textbf{رویداد داخلی \lr{(Endogenous Event)} :} خرابی ماشین آلات
		\item \textbf{رویداد خارجی \lr{(Exogenous Event)} :} تغییر در تقاضای بازار
	\end{itemize}
	\item \textbf{متغیرهای حالت \lr{(State Variables)} :} تعداد مبل‌های در دست ساخت، میزان مواد اولیه موجود
\end{itemize}
